from collections import Counter
In[36]:
genres = df['listed_in'].dropna()
all_genres = [g.strip() for sublist in genres.str.split(',') for g in sublist]
top_genres = Counter(all_genres).most_common(5)

labels = [genre for genre, count in top_genres]
sizes = [count for genre, count in top_genres]

plt.figure(figsize=(6, 6))
plt.pie(sizes, labels=labels, autopct='%1.1f%%', startangle=90)
plt.title('Top 5 Genres on Netflix')
plt.axis('equal')  
plt.show()
